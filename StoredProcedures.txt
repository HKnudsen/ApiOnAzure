CREATE OR ALTER PROCEDURE ApiOnAzureSchema.spUser_Upsert
    @FirstName NVARCHAR(50),
    @LastName NVARCHAR(50),
    @Email NVARCHAR(50),
    @Gender NVARCHAR(50),
    @UserId INT = NULL
AS
BEGIN
    IF NOT EXISTS (SELECT * FROM ApiOnAzureSchema.Users WHERE UserId = @UserId)
        BEGIN
        IF NOT EXISTS (SELECT * FROM ApiOnAzureSchema.Users WHERE Email = @Email)
            BEGIN

                INSERT INTO ApiOnAzureSchema.Users(
                    [FirstName],
                    [LastName],
                    [Email],
                    [Gender]
                ) VALUES (
                    @FirstName,
                    @LastName,
                    @Email,
                    @Gender
                )
            END
        END
    ELSE
        BEGIN
        UPDATE ApiOnAzureSchema.Users
            SET FirstName = @FirstName,
                LastName = @LastName,
                Email = @Email,
                Gender = @Gender
            WHERE UserId = @UserId
        END
END
GO

CREATE OR ALTER PROCEDURE ApiOnAzureSchema.spGet_Users
    @UserId INT = NULL
AS
BEGIN
    SELECT [Users].[UserId],
    [Users].[FirstName],
    [Users].[LastName],
    [Users].[Email],
    [Users].[Gender],
    [Orders].[OrderId],
    [Orders].[StaffId],
    [Orders].[Price],
    [Orders].[Treatment]
    FROM ApiOnAzureSchema.Users as Users
        LEFT JOIN ApiOnAzureSchema.Orders AS Orders
            ON Orders.UserId = Users.UserId
END
GO
